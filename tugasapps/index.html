Proyek A — Starter Kit
Leave Request Tracker
Sistem pengajuan cuti otomatis end-to-end
📋 Arsitektur Sistem

Form (nama, email, tanggal mulai, tanggal akhir, alasan) → Trigger onFormSubmit → catat ke sheet dengan status "Pending" → warnai kuning → kirim email ke manager → kirim konfirmasi ke karyawan. Dosen bisa approve via menu kustom → status berubah "Approved" → warnai hijau → email persetujuan terkirim.

leaveTracker.gs — Starter Kit Proyek A
// ════════════════════════════════════
// LEAVE REQUEST TRACKER — Proyek A
// Sesi 6 Capstone — UKWMS 2025
// ════════════════════════════════════

const EMAIL_MANAGER = "manager@email.com"; // ← ganti dengan email manager

// ── Menu kustom ──
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu("📅 Leave Tracker")
    .addItem("✅ Approve Pengajuan Dipilih", "approveCuti")
    .addItem("❌ Tolak Pengajuan Dipilih",   "tolakCuti")
    .addSeparator()
    .addItem("📊 Lihat Ringkasan Cuti",       "ringkasanCuti")
    .addToUi();
}

// ── Trigger: saat form disubmit ──
function prosesPermohonanCuti(e) {
  const nama       = e.values[1];
  const emailKary  = e.values[2];
  const tglMulai   = e.values[3];
  const tglAkhir   = e.values[4];
  const alasan     = e.values[5];
  const sheet      = SpreadsheetApp.getActiveSheet();
  const barisBaru  = sheet.getLastRow();

  // Tandai baris sebagai Pending (kuning)
  sheet.getRange(barisBaru, 1, 1, 6).setBackground("#FEF3C7");
  sheet.getRange(barisBaru, 7).setValue("Pending"); // kolom G = Status

  // Kirim notifikasi ke manager
  GmailApp.sendEmail(
    EMAIL_MANAGER,
    "🔔 Permohonan Cuti Baru — " + nama,
    "Ada permohonan cuti masuk:\n\n"
    + "Nama   : " + nama + "\n"
    + "Periode: " + tglMulai + " s/d " + tglAkhir + "\n"
    + "Alasan : " + alasan + "\n\n"
    + "Buka spreadsheet untuk menyetujui atau menolak."
  );

  // Kirim konfirmasi ke karyawan
  GmailApp.sendEmail(
    emailKary,
    "📋 Permohonan Cuti Diterima — Menunggu Persetujuan",
    "Halo " + nama + ",\n\nPermohonan cuti Anda sudah kami terima dan sedang diproses.\n"
    + "Periode: " + tglMulai + " s/d " + tglAkhir + "\n\n"
    + "Kami akan menghubungi Anda setelah ada keputusan.\n\nTerima kasih."
  );
}

// ── Menu: Approve baris yang dipilih ──
function approveCuti() {
  const ui    = SpreadsheetApp.getUi();
  const sheet = SpreadsheetApp.getActiveSheet();
  const baris = sheet.getActiveCell().getRow();

  if (baris < 2) { ui.alert("Pilih baris data terlebih dahulu."); return; }

  const nama      = sheet.getRange(baris, 2).getValue();
  const emailKary = sheet.getRange(baris, 3).getValue();
  const tglMulai  = sheet.getRange(baris, 4).getValue();
  const tglAkhir  = sheet.getRange(baris, 5).getValue();

  sheet.getRange(baris, 1, 1, 7).setBackground("#D1FAE5"); // hijau
  sheet.getRange(baris, 7).setValue("Approved ✓");

  GmailApp.sendEmail(
    emailKary,
    "✅ Cuti Anda Disetujui!",
    "Halo " + nama + ",\n\nPermohonan cuti Anda telah DISETUJUI.\n"
    + "Periode: " + tglMulai + " s/d " + tglAkhir + "\n\nSelamat berlibur!"
  );

  ui.alert("✅ Cuti " + nama + " berhasil disetujui. Email konfirmasi terkirim.");
}

// ── Menu: Ringkasan cuti ──
function ringkasanCuti() {
  const sheet   = SpreadsheetApp.getActiveSheet();
  const lastRow = sheet.getLastRow();
  let pending = 0, approved = 0, tolak = 0;

  for (let i = 2; i <= lastRow; i++) {
    const status = sheet.getRange(i, 7).getValue();
    if (status.includes("Pending"))   pending++;
    else if (status.includes("Approved")) approved++;
    else if (status.includes("Tolak"))    tolak++;
  }

  SpreadsheetApp.getUi().alert(
    "📊 Ringkasan Cuti\n\n"
    + "⏳ Pending  : " + pending  + " pengajuan\n"
    + "✅ Approved : " + approved + " pengajuan\n"
    + "❌ Ditolak  : " + tolak    + " pengajuan\n"
    + "──────────────────\n"
    + "Total       : " + (pending + approved + tolak) + " pengajuan"
  );
}

// ════════════════════════════════════
// MEETING MINUTES LOGGER — Proyek B
// Sesi 6 Capstone — UKWMS 2025
// ════════════════════════════════════

function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu("📝 Notulen Rapat")
    .addItem("➕ Catat Rapat Baru",           "catatRapat")
    .addItem("📧 Kirim Notulen Terakhir",      "kirimNotulen")
    .addSeparator()
    .addItem("📋 Lihat Semua Notulen",         "lihatNotulen")
    .addToUi();
}

function catatRapat() {
  const ui = SpreadsheetApp.getUi();

  // Kumpulkan info rapat via dialog bertahap
  const r1 = ui.prompt("📝 Notulen Rapat (1/4)", "Judul / Agenda Rapat:",      ui.ButtonSet.OK_CANCEL);
  if (r1.getSelectedButton() != ui.Button.OK) return;

  const r2 = ui.prompt("📝 Notulen Rapat (2/4)", "Peserta (pisahkan koma):",    ui.ButtonSet.OK_CANCEL);
  if (r2.getSelectedButton() != ui.Button.OK) return;

  const r3 = ui.prompt("📝 Notulen Rapat (3/4)", "Email peserta (pisahkan koma):", ui.ButtonSet.OK_CANCEL);
  if (r3.getSelectedButton() != ui.Button.OK) return;

  const r4 = ui.prompt("📝 Notulen Rapat (4/4)", "Poin-poin keputusan (pisahkan ;):", ui.ButtonSet.OK_CANCEL);
  if (r4.getSelectedButton() != ui.Button.OK) return;

  const judul   = r1.getResponseText();
  const peserta = r2.getResponseText();
  const emails  = r3.getResponseText();
  const poin    = r4.getResponseText();
  const waktu   = new Date().toLocaleString("id-ID");

  // Catat ke sheet
  const sheet = SpreadsheetApp.getActiveSheet();
  const baris = sheet.getLastRow() + 1;

  sheet.getRange(baris, 1).setValue(waktu);
  sheet.getRange(baris, 2).setValue(judul);
  sheet.getRange(baris, 3).setValue(peserta);
  sheet.getRange(baris, 4).setValue(poin);
  sheet.getRange(baris, 5).setValue(emails);
  sheet.getRange(baris, 1, 1, 5).setBackground("#EDE9FE"); // ungu muda

  // Format poin jadi list
  const daftarPoin = poin.split(";")
    .map(p => "• " + p.trim())
    .join("\n");

  // Kirim notulen ke semua peserta
  const daftarEmail = emails.split(",").map(e => e.trim());
  daftarEmail.forEach(email => {
    GmailApp.sendEmail(
      email,
      "📋 Notulen Rapat — " + judul,
      "Notulen Rapat\n"
      + "═══════════════════\n"
      + "Agenda  : " + judul   + "\n"
      + "Peserta : " + peserta + "\n"
      + "Waktu   : " + waktu   + "\n\n"
      + "Keputusan:\n" + daftarPoin + "\n\n"
      + "Notulen ini dibuat otomatis."
    );
  });

  ui.alert("✅ Notulen berhasil disimpan dan dikirim ke " + daftarEmail.length + " peserta.");
}

inventoryAlert.gs — Starter Kit Proyek C
// ════════════════════════════════════
// INVENTORY ALERT SYSTEM — Proyek C
// Sesi 6 Capstone — UKWMS 2025
// ════════════════════════════════════

const EMAIL_PIC   = "pengadaan@email.com"; // ← ganti dengan email PIC
const STOK_MIN    = 10; // batas minimum stok — bisa diubah

// Struktur sheet: A=Nama Barang, B=Satuan, C=Stok, D=Min, E=Status

function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu("📦 Inventory")
    .addItem("🔍 Cek Stok Sekarang",         "cekStok")
    .addItem("📧 Kirim Laporan Stok",         "kirimLaporan")
    .addSeparator()
    .addItem("🎨 Reset Warna Sheet",          "resetWarna")
    .addToUi();
}

function cekStok() {
  const sheet    = SpreadsheetApp.getActiveSheet();
  const lastRow  = sheet.getLastRow();
  let kritis = [], aman = 0;

  for (let i = 2; i <= lastRow; i++) {
    const nama   = sheet.getRange(i, 1).getValue();
    const stok   = sheet.getRange(i, 3).getValue();
    const minVal = sheet.getRange(i, 4).getValue() || STOK_MIN;
    const baris  = sheet.getRange(i, 1, 1, 5);

    if (stok <= 0) {
      baris.setBackground("#FEE2E2"); // merah — habis
      sheet.getRange(i, 5).setValue("⛔ Habis");
      kritis.push(nama + " (HABIS)");

    } else if (stok <= minVal) {
      baris.setBackground("#FEF3C7"); // kuning — hampir habis
      sheet.getRange(i, 5).setValue("⚠️ Kritis: " + stok);
      kritis.push(nama + " (sisa " + stok + ")");

    } else {
      baris.setBackground("#D1FAE5"); // hijau — aman
      sheet.getRange(i, 5).setValue("✅ Aman");
      aman++;
    }
  }

  // Kirim email jika ada stok kritis
  if (kritis.length > 0) {
    GmailApp.sendEmail(
      EMAIL_PIC,
      "⚠️ Peringatan Stok Kritis — " + kritis.length + " Barang",
      "Halo Tim Pengadaan,\n\nBerikut barang yang perlu segera dipesan:\n\n"
      + kritis.map((b, i) => (i+1) + ". " + b).join("\n")
      + "\n\nMohon segera ditindaklanjuti.\nSistem Inventory Otomatis"
    );
  }

  SpreadsheetApp.getUi().alert(
    "📊 Hasil Cek Stok\n\n"
    + "✅ Aman    : " + aman          + " barang\n"
    + "⚠️ Kritis  : " + kritis.length + " barang"
    + (kritis.length > 0 ? "\n\nEmail peringatan sudah dikirim ke PIC." : "")
  );
}

function resetWarna() {
  const sheet   = SpreadsheetApp.getActiveSheet();
  const lastRow = sheet.getLastRow();
  for (let i = 2; i <= lastRow; i++) {
    sheet.getRange(i, 1, 1, 5).setBackground(null);
    sheet.getRange(i, 5).clearContent();
  }
  SpreadsheetApp.getUi().alert("✅ Sheet berhasil direset.");
}
